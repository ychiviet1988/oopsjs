  5 //Define a class
  6 /**
  7  * Employee class to represent employee which has employee id and name as private properties.	
  8  * 
  9  * @class Employee
 10  * @author Rajendra Patil <rrpatil26@gmail.com>
 11  * 
 12  */
 13 var Employee = function(){
 14 	
 15 		//private variables	- actually protected - will be inherited
 16 		/**
 17 		 * @private _name
 18 		 */
 19 		var _name;
 20 		/**
 21 		 * @private _employeeId
 22 		 */
 23 		
 24 		var _employeeId;
 25 
 26 		//private methods - actually protected - will be inherited
 27 		/**
 28 		 * @private _setName
 29 		 * @param name - the name to be set
 30 		 */
 31 		var _setName = function(name){
 32 			_name = name;
 33 		} 
 34 		
 35 		/**
 36 		 * @private _getName
 37 		 * @returns - the name of an employee
 38 		 */
 39 		var _getName = function(){
 40 			return _name;
 41 		}
 42 		
 43 		//public methods
 44 		/** This is kind of constructor to initialize the object properties. 
 45 		 * @public
 46 		 * @constructs
 47 		 * @param name - name of an employee
 48 		 * @param id - id of an emlployee
 49 		 * @returns this - init must always return this
 50 		 */
 51 		this.init = function(name,id){
 52 			_name = name;
 53 			_employeeId = id;
 54 			return this;
 55 		}
 56 		/** Public method to get name of an employee. 
 57 		 * @public
 58 		 * @returns name 
 59 		 */
 60 		this.getName = function(){
 61 			return _name;
 62 		}
 63 
 64 		/** Public method to get employee id of an employee. 
 65 		 * @public
 66 		 * @returns employeeId 
 67 		 */
 68 		this.getEmployeeId = function(){
 69 			return _employeeId;
 70 		}
 71 
 72 		/** Public method to to represent this class as a string
 73 		 * @public
 74 		 * @returns string - string representation of an employee
 75 		 */
 76 		this.asString = function(){
 77 			return "Employee { name: "+_name+", id: "+_employeeId+" }";
 78 		}
 79 };
 80 /**
 81  * Manager class to represent an employee which has reportees too.	
 82  * this class is herited from Employee
 83  * @see Employee
 84  * @class Manager
 85  * @author Rajendra Patil <rrpatil26@gmail.com>
 86  * @extends Employee 
 87  */
 88 var Manager = function(){
 89 
 90 		//own private var	
 91 		/** @private */
 92 		var _reportees = []; 
 93 
 94 		//own private methods
 95 		/** @private */
 96 		var _addReportee = function(employee){
 97 			_reportees.push(employee);
 98 		};
 99 
100 		//own public methods
101 		//Constructor
102 		/** This is kind of constructor to initialize the object properties. 
103 		 * @public
104 		 * @constructs
105 		 * @param name - name 
106 		 * @param id - employee id 
107 		 * @returns this - init must always return this
108 		 */
109 		this.init = function(name,id){
110 			_super.init(name,id);//_super points to parent
111 			_reportees = [];
112 			return this;	
113 		};
114 		//Method overridding
115 
116 		/** Public method to to get manager name - overrided from Employee
117 		 * @see Employee#getName
118 		 * @public
119 		 * @returns name - manager name
120 		 */
121 		this.getName = function(){
122 			return _super._getName(); //calling super's private method
123 		};
124 
125 		/** Public method to to get list of reportees 
126 		 * @public
127 		 * @returns reportees - array of employees who are reporting to this manager
128 		 */
129 		this.getReportees = function(){
130 			return _reportees;
131 		}
132 		/** Public method to to represent this class as a string - overrided from Employee
133 		 * @see Employee#asString
134 		 * @public
135 		 * @returns string - string representation of an Manager
136 		 */
137 		this.asString = function(){
138 			return "Manager { name: "+_name+", id: "+_employeeId+" }";
139 		}
140 
141 }.inherits(Employee);
